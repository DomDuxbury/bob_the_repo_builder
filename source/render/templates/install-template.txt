#!/bin/bash

# Create virtualenv
source $HOME/.local/bin/virtualenvwrapper.sh
mkvirtualenv {{ name }}

# Install flake8
pip install flake8

# Install pytest
pip install pytest

# Create requirements.txt
pip freeze > requirements.txt

# Initialise git
git init

# Install pre-commit hook
rm .git/hooks/pre-commit
ln pre-commit.sh .git/hooks/pre-commit
